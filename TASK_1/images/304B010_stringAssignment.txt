package Datatypes;

import java.util.Locale;

public class Main {

    public static void main(String[] args) {
	String s1="Java";
	String s2="Programming";
        System.out.println("String 1 = "+s1);
        System.out.println("String 2 = "+s2);
        System.out.println("length of s1 = "+ s1.length());
        System.out.println("length of s2 = "+ s2.length());
        System.out.println("concatenation => "+s1.concat(s2));
        System.out.println("Uppercase conversion => "+ s1.toUpperCase());
        System.out.println("lowercase conversion => "+ s1.toLowerCase());
        System.out.println("Modified s2 => "+ s2.replace('m','z'));
        System.out.println("character at index 3 in s2 => "+ s2.charAt(3));
        System.out.println("comparison using compareTo() => "+ s1.compareTo(s2));
        System.out.println("comparison using equals() = "+ s1.equals(s2));
        System.out.println("index of 'v' => "+s1.indexOf('v'));
        System.out.println("Checking s2 is empty or not => "+s2.isEmpty());
    }
}
